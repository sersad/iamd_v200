CREATE TABLE playlist (
    id      INTEGER PRIMARY KEY AUTOINCREMENT
                    UNIQUE,
    name    VARCHAR UNIQUE ON CONFLICT ROLLBACK,
    url     VARCHAR NOT NULL,
    status  INTEGER    DEFAULT (False),
    bitrate INTEGER
);

CREATE TABLE image (
    id    INTEGER  REFERENCES playlist (id) ON DELETE CASCADE
                                        ON UPDATE CASCADE,
    image BLOB
);
